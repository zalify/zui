## Usage

```tsx
import { RadioCardGroup } from '@zalify-ui/react'
```

```tsx
<RadioCardGroup.Root>
  <RadioCardGroup.Label />
  <RadioCardGroup.Item>
    <RadioCardGroup.ItemHiddenInput />
    <RadioCardGroup.ItemText />
    <RadioCardGroup.ItemControl />
  </RadioCardGroup.Item>
</RadioCardGroup.Root>
```
