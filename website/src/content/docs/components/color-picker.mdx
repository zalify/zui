## Usage

```tsx
import { ColorPicker } from '@zalify-ui/react'
```

```tsx
<ColorPicker.Root>
  <ColorPicker.Control>
    <ColorPicker.ChannelInput channel="hex" />
    <ColorPicker.Trigger>
      <ColorPicker.Swatch value="#000000" />
    </ColorPicker.Trigger>
  </ColorPicker.Control>
</ColorPicker.Root>
```
