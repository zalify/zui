## Usage

```tsx
import { Grid, GridItem } from '@zalify-ui/react'
```

```tsx
<Grid columns={4} gap="4">
  <GridItem colSpan={2}>
    <div>Spans 2 columns</div>
  </GridItem>
  <GridItem colSpan={1}>
    <div>Spans 1 column</div>
  </GridItem>
  <GridItem colSpan={1}>
    <div>Spans 1 column</div>
  </GridItem>
</Grid>
```
